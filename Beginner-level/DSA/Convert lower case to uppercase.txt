// Program To Convert Uppercase To Lowercase in CPP by Sooryaraj
#include <cctype>
#include <iostream>
#include <cstring>
using namespace std;
int main() {
  char str[] = "Hello I am Soorya";
  char ch;
  cout << "The lowercase String for \"" << str << "\" is " << endl;
  for (int i = 0; i < strlen(str); i++) {
    ch = tolower(str[i]);
    cout << ch;
  }
  return 0;
}
